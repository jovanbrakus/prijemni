\[ f_{\max} = (a - 1) - \frac{(-a)^2}{4a} \]

\( f_{\max} = \dfrac{3 \cdot (-2) - 4}{4} = \dfrac{-6 - 4}{4} = \dfrac{-10}{4} = -2{,}5 \)
